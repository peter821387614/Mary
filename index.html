import java.util.ArrayList; import java.util.List; public class Hanoi2 { private List moves = new ArrayList(); public List solve(int n) { move(n, "A", "B", "C"); return moves; } private void move(int n, String start, String middle, String end) { if (n == 1) { moves.add(new Move(start, end)); } else { move(n - 1, start, end, middle); moves.add(new Move(start, end)); move(n - 1, middle, start, end); } } } class Move { public String start; public String end; public Move(String start, String end) { this.start = start; this.end = end; } }
